#$ TITLE: PID Tune Attempt 2 (Fixed?)
#$ FIRMWARE_VERSION: 4.5
#$ CATEGORY: TUNE
#$ STATUS: EXPERIMENTAL
#$ DESCRIPTION: tuning for stability without shaking
#$ DESCRIPTION: reduced integral gain to prevent windup
#$ FORCE_OPTIONS_REVIEW: TRUE

#$ INCLUDE: presets/4.5/tune/defaults.txt

# proportional gain set to standard values to stop shaking
# enough force to correct errors without oscillation
set p_pitch = 45
set p_roll = 45
set p_yaw = 45

# integral gain set very low to minimize windup risk
# maintains safety but might reduce holding power
set i_pitch = 15
set i_roll = 15
set i_yaw = 15

# derivative gain increased to smooth out the motion
# higher damping prevents jitter but adds motor load
set d_pitch = 45
set d_roll = 45
set d_yaw = 0

# feedforward increased for better stick tracking
# keeps the response feeling connected
set f_pitch = 60
set f_roll = 60
set f_yaw = 60

# anti gravity remains off for natural feel
set anti_gravity_gain = 0

# iterm relax set to setpoint to handle stick moves
set iterm_relax = RP

set throttle_boost = 5
set dyn_idle_min_rpm = 0

simplified_tuning apply
